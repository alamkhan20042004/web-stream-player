<body>

    <!-- Naya Iframe add karne ka form -->
    <div class="add-stream-container">
        <span>➕ Add Channel:</span>
        <input type="text" id="streamName" placeholder="Channel Name (e.g. Sports HD)">
        <input type="text" id="streamUrl" placeholder="Iframe Link (e.g. https://...)">
        <button class="add-btn" onclick="addNewStream()">Add to List</button>
    </div>

    <!-- Channels ki list jahan buttons show honge -->
    <div class="channel-list" id="channelList">
        <!-- Default button -->
        <button class="channel-btn active" onclick="changeStream(this, 'https://embedhd.org/source/fetch.php?hd=71')">Channel HD 71</button>
    </div>

    <!-- Main Player -->
    <div class="player-container">
        <iframe id="mainPlayer" src="https://embedhd.org/source/fetch.php?hd=71" allowfullscreen allow="encrypted-media; picture-in-picture; fullscreen"></iframe>
    </div>

    <script>
        // Stream change karne ka function
        function changeStream(buttonElement, url) {
            // Player ka link change karo
            document.getElementById('mainPlayer').src = url;

            // Purane active button se color hatao
            const allButtons = document.querySelectorAll('.channel-btn');
            allButtons.forEach(btn => btn.classList.remove('active'));

            // Naye button ko active color do
            buttonElement.classList.add('active');
        }

        // Naya stream button add karne ka function
        function addNewStream() {
            const nameInput = document.getElementById('streamName').value;
            const urlInput = document.getElementById('streamUrl').value;

            if (nameInput === '' || urlInput === '') {
                alert('Bhai, Channel Name aur Link dono enter karo!');
                return;
            }

            // Naya button create karo
            const newBtn = document.createElement('button');
            newBtn.className = 'channel-btn';
            newBtn.textContent = nameInput;
            
            // Button click par stream change ho
            newBtn.onclick = function() {
                changeStream(this, urlInput);
            };

            // List mein add karo
            document.getElementById('channelList').appendChild(newBtn);

            // Inputs ko clear karo
            document.getElementById('streamName').value = '';
            document.getElementById('streamUrl').value = '';
        }
    </script>
